71  normalizedCol.append(self.createCol(col, self.normalizeData[col_idx]))
72  algorithm_data = ufunc.hstack(normalizedCol)

74  ufunc.parameters_reduction(algorithm_data)

265  def transferToRange(self, algorithm_data):
266  """
267  Transfer value from real world range to normalized range
268 
269  The ranges are specified in the NormalizedData and inserted to the AlgorithmData which is
270  the parameter for this method
271 
272  Args,
273  algorithm_data , (AlgorithmData) - The algorithm_data to normalize
274 
275  Returns,
276  AlgorithmData , The normalized column
277  """
278  for col_idx in range(algorithm_data.shape[1]):
279  colMin = algorithm_data[RowNames.Min, col_idx]
280  colMax = algorithm_data[RowNames.Max, col_idx]
281  targetMin = algorithm_data[RowNames.MinTarget, col_idx]
282  targetMax = algorithm_data[RowNames.MaxTarget, col_idx]
283  if colMin == colMax:
284  for row_idx in range(len(algorithm_data)):
285  algorithm_data[row_idx, col_idx] = targetMin
286  else:
287  for row_idx in range(len(algorithm_data)):
288  algorithm_data[row_idx, col_idx] = (
289  (targetMax - targetMin) *
290  (algorithm_data[row_idx, col_idx] - colMin)) / (colMax - colMin) + targetMin
291  return algorithm_data

293  def createEquilateralMatrix(self, n):
294  """
295  Create an Equilateral matrix
296  Args,
297  n , (int) - The number of values that the matrix is for
298 
299  Returns,
300  List[List{float]] , The equilateral matrix
301  """
302 
303  result = [[0 for i in range(n - 1)] for j in range(n)]
304  result[0][0] = -1
305  result[1][0] = 1.0
306  for k in range(2, n):
307  r = k
308  f = math.sqrt(r * r - 1.0) / r
309  for i in range(k):
310  for j in range(k - 1):
311  result[i][j] *= f
312 
313  r = -1.0 / r
314  for i in range(k):
315  result[i][k - 1] = r
316 
317  result[k][k - 1] = 1.0
318 
319  return result
