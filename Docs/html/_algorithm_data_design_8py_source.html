89  def loadNormalizeData(self, rowDataMatrix):
90  """
91  Load normalize data from a file and fit the data matrix to the normalize data
92 
93  - Each row in the normalize data represent a column in the data matrix
94  - there are 2 possibilities for a column
95  -# The column is representing a column in the data matrix : In this case
96  the column will be copied from the row data matrix to the target data matrix
97  according to the column number in the normalize data
98  -# The column is not part of the data matrix : In this case the column number in the
99  normalize data will be -1 and the method will add an empty column to the data matrix
100 
101  Args :
102  rowDataMatrix : ndarray with 2 dimensions (this parameter is for use of multipile activation
103  of the method for different arrays (specifically algorithm data and test data))
104 
105  Returns:
106  bool : if the load from the file succeeded
107  normalizeData : the normalize data matrix
108  ndarray 2d : the converted data matrix
109  """
110  dataMatrix = []
111  try:
112  normalizeData = Parameter.from_csv(self.designFilename)
113  except Exception as e:
114  QMessageBox.information(self, "AlgorithmDataDesign dialog",
115  "Failed to load from file error is : \n" +
116  str(e) + "\nGenerating new")
117  return False, None, rowDataMatrix
118 
119  for nd in normalizeData:
120  if (nd["indexInDataFile"] == -1):
121  newCol = np.full(
122  rowDataMatrix.shape[0], "0", dtype=np.dtype(('U100', 1)))
123  newCol[0] = nd["fieldName"]
124  dataMatrix.append(newCol)
125  else:
126  dataMatrix.append(rowDataMatrix[:, nd["indexInDataFile"]])
127  dataMatrix = np.stack(dataMatrix, axis=-1)
128 
129  return True, normalizeData, dataMatrix

301  def swapRows(self, firstRow: int, secondRow: int):
302  """
303  Swap between 2 rows (used in Up and Down)
304  """
305 
306  # Create 2 temp Normalize input
307  firstTemp = NormalizeInput(self, self.dataMatrix,
308  NormalizeData(-1, "Temp"),
309  self.gridLayout_fields, -1)
310  secondTemp = NormalizeInput(self, self.dataMatrix,
311  NormalizeData(-1, "Temp"),

313 
314  # Replace the existing rows with the temp rows
315  self.normalizeInputList[firstRow].replaceInGrid(firstTemp, False)
316  self.normalizeInputList[secondRow].replaceInGrid(secondTemp, False)
317 
318  # Replace the temp rows with the existing rows (in reverse order)
319  firstTemp.replaceInGrid(self.normalizeInputList[secondRow], True)
320  secondTemp.replaceInGrid(self.normalizeInputList[firstRow], True)
321 
322  # swap the rows in the NormalizedInputList
323  self.normalizeInputList[firstRow], self.normalizeInputList[secondRow] = \
324  self.normalizeInputList[secondRow], self.normalizeInputList[firstRow]
325 
326  # Update the rowIndex variable and the matrix in the NormalizeInput
327  self.updateNormalizeInput()
328 
329  # swap the columns in the dataMatrix
330  self.dataMatrix[:, [firstRow, secondRow
331  ]] = self.dataMatrix[:, [secondRow, firstRow]]
