327  def transformCharPath(self):
328  """ Transform char path when the item is setted

348  def fillOccupied(self):
349  """ Fill the occupied matrix
350 
351  The algorithm of filling the occupied matrix is
352  -# Scanning the char path
353  -# For each point decide on where row and column of the net
354  -# Set the occupies matrix for this column and row to True
355  """
356 
357  for idx in range(100):
358  point = self.charPath.path().pointAtPercent(idx / 100.)
359  row_idx, col_idx = self.calcRowCol(point)
360  self.occupied[row_idx][col_idx] = True
361 
362  def calcRowCol(self, point: QPointF):
363  """ Calculate the network row and column that a point is int
364  calc the row and column indexes of a point
365 
366  The following is the algorithm:
367  1. Find the distance between the point and the left (or top)
368  2. Divide the distance with the width of path to find the relative position
369  3. Multipile this relative position with the number of rows/cols
370  4. Convert the result to int to find the indexes
371  5. If the index is the number of row/col reduce the index
372  (This is for the case the the point is on the boundary and in this
373  case the relative position is 1 which will cause the indexes to
374  be the number of rows/cols - out of the matrix indexes)
375 
376  Args:
377  point (QPointF) : The point to resolve
378 
379  Returns:
380  int : The network row that the point is in
381  int : The network column that the point is in
382 
383  """
384  partialX = (point.x() - self.charPath.boundingRect().left()) / self.charPath.boundingRect().width()
385  partialY = (point.y() - self.charPath.boundingRect().top()) / self.charPath.boundingRect().height()
386  col_idx = int(partialX * self.netCols)
387  row_idx = int(partialY * self.netRows)
388  if row_idx == self.netRows:
389  row_idx -= 1
390  if col_idx == self.netCols:
391  col_idx -= 1
392  return row_idx, col_idx
