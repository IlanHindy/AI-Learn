78 class RunningDialogModel(object):
79  """
80  Process methods of an algorithm according to requests of the dialog
81 
82  ## The sequence of processing an algorithm:
83  -# Load - load the data from csv files to self.data_matrix self.test_data_matrix
84  -# Normalize - Show a dialog for designing the algorithm data and produce a normalized
85  self.algorithm_data and self.test_algorithm_data according to the design
86  -# Init - Activate the init method of the train
87  -# Step/Run to end - Process the algorithm
88  -# Test - test the algorithm with test data
89  -# Scipy - Activate the scipy version of the algorithm
90  -# ScikitLearn - Activate the scikit - Learn of the algorithm
91 
92  ## Operating method
93  -# The run method is activated by the dialog with parameter that tells which processing step is requested
94  -# The run method activate a thread with thread_main as its starting point
95  -# The thread_main method activates the requested method (for example load, init, step etc.)
96  -# The requested method returns 4 results which are set as members of the module
97  -# self.success - If the processing succeeded
98  -# self.result - The result of the method
99  -# self.end_message - A return message from the method
100  -# self.algorithm_message - A return message from the algorithm
101  -# After getting the results the thread_main set an event to signal to the dialog signaling that it finished
102 
103  ## Summary of connections between the dialog and the model
104  ### Variables created by the dialog
105  #### Variables used for all the actions
106  -# self.view.event - The event is initialized by the dialog and given to model in the run method.
107  The event is set by the model to signal to the dialog that the processing ended
108  -# self.view.messages - A queue with messages from the model to be presented on the dialog
109 
110  #### Variables used for the load action
111  -# self.train - The training class name (which is converted by the method to an object from the class)
112  -# self.algorithm - The algorithm class name (which is converted by the method to an object from the class)
113  -# self.filename - The file name to load the algorithm from
114  -# self.test_filename - The file name to load the test data
115 
116  #### Variables used by the normalized actions
117  -# self.normalize_data - The normalize data as designed by the AlgorithmDataDesign dialog
118  -# self.data_matrix - The data matrix as generated by the AlgorithmDataDesign dialog
119  -# self.test_data_matrix - The test data matrix as generated in the dialog using the same AlgorithmDataDesign dialog
120 
121 
122  #### Variables used in the step/run_to_end actions
123  -# self.num_steps - The maximum number of steps to process in one step method
124  or the number of steps between 2 updates of the plot in run to end
125 
126  #### Variables used in the test action
127  - No new variables are needed for this action from the dialog
128 
129  #### Variables used in the scipy action
130  - No new variables are needed for this action from the dialog
131 
132  #### Variables used in the scikitLearn action
133  - No new variables are needed for this action from the dialog
134 
135  ### Variables created by the model
136  -# self.success - Informs the dialog if the algorithm succeeded
137  -# self.result - The result of the running
138  -# self.end_message - A message produced by the model to inform ending of the processing
139  -# self.algorithm_message - A message produced by the algorithm informing the result of the processing
140  -# self.enable_actions - A dictionary that tells the dialog which actions are possible after processing
141  the current step
142 
143  """

183  def run(self, action : Actions, event: Event):
184  """
185  Run a step in the processing (The public method of the model)

308  def step(self):
309  """
310  Process step/steps in the training
311 
312  #### The following are the possible conditions to stop the loop
313  -# The number of cycles set by the dialog ended
314  -# An error occurred in the processing of the step in the train/algorithm
315  -# The algorithm finished (The check method of the train returned True)
316 
317  Returns:
318  bool : False if error True if ok
319  bool : If the algorithm finished
320  string : The step end message
321  string : The result message from the algorithm
322  """
323  end_idx = self.step_idx + self.num_steps
324  while self.step_idx < end_idx:
325  success, message, new_score, finished = self.train.step()
326  if not success:
327  return False, False, "The step failed ", message
328 
329  if message != "":
330  if new_score:
331  self.view.messages.put(Message(Qt.yellow, str(self.step_idx) + ". " + message), block = False)
332  else:
333  self.view.messages.put(Message(Qt.white, str(self.step_idx) + ". " + message), block = False)
334 
335 
336  if new_score:
337  self.view.plot()
338 
339  if finished:
340  self.enable_actions[Actions.Step] = False
341  self.enable_actions[Actions.RunToEnd] = False
342  self.enable_actions[Actions.Test] = True
343  return True, True, "Finished algorithm after " + str(self.step_idx) + " steps", ""
344 
345  self.step_idx += 1
346 
347  return True, False, "Finished " + str(self.step_idx) + " steps -- did not finished", ""

349  def run_to_end(self):
350  """
351  Process the algorithm until finished
352 
353  This method activates the step method of the train in an infinite loop
354 
355  #### The following are the possible conditions to stop the loop
356  -# An error occurred in the processing of one of these actions
357  -# The algorithm finished (The check method of the train returned True)
358 
359  Returns:
360  bool : False if error True if finished
361  bool : False if error True if finished
362  string : The run to end - end message
363  string : The result message from the algorithm
364 
365  """
366 
367  # Perform until end of the algorithm
368  while True:
369 
370  # Perform the step and return if failed
371  success, message, new_score, finished = self.train.step()
372  if not success:
373  return False, False, "The step() failed error message :", message
374 
375  if message != "":
376  if new_score:
377  self.view.messages.put(Message(Qt.yellow, str(self.step_idx) + ". " + message), block = False)
378  else:
379  self.view.messages.put(Message(Qt.white, str(self.step_idx) + ". " + message), block = False)
380 
381  if new_score:
382  self.view.plot()
383 
384  if finished:
385  self.enable_actions[Actions.Step] = False
386  self.enable_actions[Actions.RunToEnd] = False
387  self.enable_actions[Actions.Test] = True
388  return True, True, "Finished algorithm after " + str(self.step_idx) + " steps", ""
389 
390  self.step_idx += 1

488 class RunningDialog(QDialog, Ui_RunningDialog):
489  """
490  running dialog for all the algorithms
491 
492  # Dialog Description
493 
494  ## Purpose
495  To run an AI algorithms
496 
497  ## Structure
498  The dialog is composed from
499  -# Horizontal layout that holds 2 Plot containers (viewing of a graph)
500  and a text edit for the messages
501  -# Grid layout with 4 rows (from top to button):
502  -# labels that will be used to show the time elapse for the action activated by the
503  buttons below them
504  -# The buttons row to activate the algorithm
505  -# labels that will be used to show the time elapse for independent actions
506  -# The buttons row for the independent actions
507 
508  ## Dialog working method
509  -# Each one of the buttons is responsible for one step in the processing of the
510  algorithm
511  -# The steps are :
512  -# Load
513  -# Normalize
514  -# Init
515  -# Repeating steps or Run to end
516  -# Test
517  -# Additionally There are independent actions:
518  -# Activate the scipy version of the algorithm
519  -# Activating the scikit - Learn version of the algorithm
520  -# Each button is doing:
521  -# Get all the inputs needed for the action
522  -# Activate the activate method
523  -# The activate method does:
524  -# Start a timer
525  -# Call the run method of the model to perform the action
526  -# While the action is running in a separate thread in the model, the timer is active in the dialog
527  -# The timer is running as long as the model is running an action
528  -# When the model finished an action, it sets an event
529  -# The timer method checks the event and if it is set it does:
530  -# Stop the timer
531  -# Generate a finish/error message
532  -# Set the enable attribute of the buttons according a dictionary in the model
533 
534  ## Attributes
535  -# self.model - The model of the dialog (That performs the actual operations)
536  -# self.timer - The timer that is presenting the time and activates the end action operations
537  -# self.action - The action to perform (set by the activate method to be sent to the model)
538  -# self.timerLabel - Tells the timer event handler where to write the time elapsed when an action is performed
539  -# self.timerText - A text for the time label (set by the action method and used by the timer event handler)
540  -# self.startTime - The start time of the action to be used by the timer event
541  -# self.stopEvent - The event that will be used by the model to signal to the timer that it finished
542 
543  Args:
544  Parent : (QDialog) - The calling dialog
545  """
