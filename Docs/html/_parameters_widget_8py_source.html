68 class ParametersList(object):

70  This class holds a list of standared parameters and implements some utilities on them

72  Access to the parameters:

81  def __init__(self, parameters: List[StdPrm]):
82  # remove the duplications from the list
83  # insert the indexes of the duplications to a list
84  to_remove = []
85  for idx in range(len(parameters)):
86  name = parameters[idx]["name"]
87  for sIdx in range(idx + 1, len(parameters)):
88  if parameters[sIdx]["name"] == name:
89  to_remove.append(sIdx)
90 
91  # generate a list only from the elements which indexes are not found in the list
92  self.parameters = [parameters[idx] for idx in range(len(parameters)) if not idx in to_remove]

101  """
102  get an item by index

104  There are 3 types of indexes:

112  Returns:
113  Any: The value of the parameter

154  def load(self, clss: str, filename: str) -> Tuple[bool, str]:
155  """
156  Loads a csv file and update the parameters
157 
158  The following is the structure of the file
159  - A key line
160  - The values of a new attribute stating the class
161  - The values of the rest of the parameters
162  - The method reads only the first 2 column of each line
163  - The first line which contains the keys of the parameters is skipped
164  - The class of the file is checked (with the parameter) if they are not the same - an error returned
165  - For each of the rest of the lines the suitable parameter (identified by the name which is the first column) is updated
166 
167  Args:
168  clss: (str) - The name of the class
169  filename : (str) - The file name
170 
171  Returns:
172  bool - If the algorithm is the same one as the one presented in the dialog
173  str - Error message
174 
175  """
176 
177  # load the csv
178  lines = FileUtiles.csvLoad(filename)
179 
180  # check if the file is from the same algorithm as the selected algorithm
181  # the first line holds the algorithm's name
182  if clss != eval(lines[1][1]):
183  return False, "The current class is " + clss + " and the file is for :" + lines[1][1]
184 
185  # load the parameters
186  # the parameters starts from line 2
187  for line in lines[2:]:
188  self[eval(line[0])]["value"] = "" if (line[1] == "") else eval(line[1])
189 
190  return True, ""
191 
192  def finish(self) -> Tuple[bool, str]:
193  """
194  - This method checks if all the parameters got leagal values
195  - The check is done by activating the check method of each parameter
196 
197  Returns:
198  bool: If all the check succeeded
199  str: The error message
200  """
201  for std_prm in self.parameters:
202  result, message = std_prm.check()
203  if not result:
204  return result, message
205  return True, ""
