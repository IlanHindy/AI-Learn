21  """
22  This class implements the KMeansClustering algorithm
23 
24  # Algorithm Description
25  ## Perpose:
26  To divide observations to groups
27 
28  ## Algorithm parameters
29  - The number of groups to divide the observations to
30  - The observations
31 
32  ## Algorithm principals:
33  - Each group has a centroid the centroid is a vector with the size of the
34  number of the input fields fields
35  - The initiation is to put one observation (randomly selected) as a centroid of each group
36  - The algorithm works in iterations. in each iteration there are 2 steps:
37  - Put each observation at the group which it's centroid is the nearest to
38  - Calculate the new centroid as the avarage of the members in the group
39  - The algorithm finishes when there are no transfers of observations from one group to another
40 
41  ## Attributes
42  - algorithm_data (AlgorithmData) : holds all the data for processing an algorithm
43  - finished_alg (bool) : whether the algorithm ended
44  - num_groups (int) : the number of groups
45  - numFields (int) : the number of fields in each observation
46  - centroids (num_groups X numFields float matrix)
47 
48  Args:
49  parameters (ParametersList) : a list of parameters to the algorithm
50  algorithm_data (AlgorithmData) : a normalized AlgorithmData
51  """

132  @staticmethod
133  def find_centroid(num_groups: int, algorithm_data: AlgorithmData, centroids: List[List[int]]) -> bool:
134  """
135  Find and put each observation in the new group
136 
137  This method is used in 2 cases
138  -# At the learning (clustering) phase
139  -# At the testing phase
140 
141  In each case a different algorithm_data is used
142 
143  Args:
144  num_groups : (int) - The number of groups in the algorithm
145  algorithm_data : (AlgorithmData) - The algorithm_data to process
146  centroids : (ndarray matrix of float) - The centroids
147 
148  Returns:
149  bool : whether there was a change in the groups
150 
151  """
152  dist_met = DistanceMetrics()
153  finished = True
154 
155  # from each observation find the neerest centroid
156  for row_idx in range(algorithm_data.shape[0]):
157  selected_centroid = -1
158  min_dist = float('inf')
159  parameters = algorithm_data[row_idx, FieldRolls.Parameter]
160 
161  # for each group
162  for group_idx in range(num_groups):
163 
164  # calc the distance between the observation and the centroid
165  dist = dist_met.euclidean(centroids[group_idx], parameters)
166 
167  # if the distance is smaller then the previous one set the group
168  if (dist < min_dist):
169  min_dist = dist
170  selected_centroid = group_idx
171 
172  # After there is a dissition for the observation update set the group and update
173  # the finished flag
174  if (selected_centroid != algorithm_data[row_idx, FieldRolls.StepResult]):
175  finished = False
176  algorithm_data[row_idx, FieldRolls.StepResult] = selected_centroid
177 
178  return finished

180  def calcCentroids(self):
181  """
182  Calculate the new centroid after each step
183 
184  This method does 2 things:
185  -# After the groups where constructed calc the new centroid as the avarage of the
186  observations in the group
187  -# Calc the avarage of the distance between the observations and the new centroid
188  for presentation perposes
189  """
190  dist_met = DistanceMetrics()
191  self.evaluation = 0
192 
193  # calc the centroid
194  for group_idx in range(self.num_groups):
195 
196  # Generate a list of rows bellongs to the group
197  members = np.where(self.algorithm_data[:, FieldRolls.StepResult] == group_idx)[0]
198 
199  # generate a matrix of members X the members parameters
200  members_parameters = self.algorithm_data[members.tolist(), FieldRolls.Parameter]
201 
202  # calculate the avarage of each parameters and put it in the centroid
203  for parameter_idx in range(self.num_fields):
204  self.centroids[group_idx, parameter_idx] = np.average(members_parameters[:, parameter_idx])
205 
206  # Create a vector holds the distances between the observations and the new centroid
207  members_dist = np.zeros(members.shape[0])
208  for member_idx in range(members.shape[0]):
209  members_dist[member_idx] = dist_met.euclidean(members_parameters[member_idx], self.centroids[group_idx])
210 
211  # Calc the avarage of the distances and add it to the evaluation
212  self.evaluation += np.average(members_dist)

214  @staticmethod
215  def create_test_conv(num_groups: int, algorithm_data: AlgorithmData) -> np.ndarray:
216  """
217  Create conversion between the groups as found in the algorithm and the groups that are in the expected result
218 
219  In order to see if the algorithm is working we have to convert between the groups
220  that where used in the algorithm (which where randomly created in the init phase)
221  and the groups used in the result
222 
223  The method of the conversion is to see for each group generated by the algorithm, which
224  is the group from the result groups that has the largest number of members in the algorithm group
225 
226  Args:
227  num_groups : (int) - The number of groups
228  algorithm_data : (AlgorithmData) - The algorithm data
229 
230  Returns:
231  numpy vector : The convertion method
232 
233  """
234 
235  # The test conv is converting between the number of the group in the centroid
236  # (The location in the vector) and the group in the result (the value in the vector)
237  test_conv = np.zeros(num_groups)
238  for step_group_idx in range(num_groups):
239 
240  # Get all the rows with the step_group_idx in the stepResult column
241  rows_of_step_group = np.argwhere(
242  algorithm_data[:, FieldRolls.StepResult] == step_group_idx).flatten().tolist()
243 
244  # Extract from the matrix with the rows with the step_group_idx the column with the result group
245  step_group_data = algorithm_data[rows_of_step_group, FieldRolls.ResultPresentation]
246 
247  result_group_size = 0
248 
249  # Find the result group with the largest number of entries in the vectors
250  # This result group is the group that the algorithm claims that is the step group
251  for result_group_Idx in range(num_groups):
252  num_result_in_group = np.where(step_group_data == result_group_Idx)[0].shape[0]
253  if num_result_in_group > result_group_size:
254  result_group_size = num_result_in_group
255  test_conv[step_group_idx] = result_group_Idx
256 
257  return test_conv
